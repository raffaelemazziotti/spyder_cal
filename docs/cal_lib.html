 20    """
 21    A class to interface with the SpyderX colorimeter for monitor calibration.
 22
 23    This class provides methods for initializing the SpyderX device, performing black calibration,
 24    measuring luminance values, and extracting factory calibration data. The SpyderX communicates
 25    through USB, requiring `libusb` as the backend.
 26
 27    Attributes:
 28        dev (usb.core.Device): The USB device representing the SpyderX.
 29        spyderData (dict): A dictionary storing SpyderX calibration and measurement data.
 30        backend (usb.backend): The USB backend used for communication.
 31    """
 32
 33    def __init__(self, libusb_path):
 34        """
 35        Initializes the SpyderX class.
 36
 37        Args:
 38            libusb_path (str): Full path to the `libusb-1.0.dll` file for Windows.
 39        """
 40        self.dev = None
 41        self.spyderData = {}
 42        self.backend = usb.backend.libusb1.get_backend(find_library=lambda x: libusb_path)
 43        if self.backend is None:
 44            raise ValueError("Libusb backend not found. Check if the path is correct.")
 45
 46    def initialize(self):
 47        """
 48        Initializes the SpyderX device and performs initial USB setup.
 49
 50        Returns:
 51            bool: True if the device was successfully initialized, False otherwise.
 52        """
 53        try:
 54            self.dev = usb.core.find(idVendor=0x085C, idProduct=0x0A00, backend=self.backend)
 55            if self.dev is None:
 56                print("SpyderX device not found. Is it plugged in?")
 57                return False
 58
 59            # Set USB configuration
 60            try:
 61                self.dev.set_configuration()
 62            except NotImplementedError:
 63                print("Set configuration not supported on this platform.")
 64            except usb.core.USBError as e:
 65                print(f"Error setting configuration: {e}")
 66
 67            # Claim USB interface
 68            try:
 69                if self.dev.is_kernel_driver_active(0):
 70                    self.dev.detach_kernel_driver(0)
 71                usb.util.claim_interface(self.dev, 0)
 72            except NotImplementedError:
 73                print("Claim interface not supported on this platform.")
 74            except usb.core.USBError as e:
 75                print(f"Error claiming interface: {e}")
 76
 77            # Perform setup transfers
 78            self._control_transfer(0x02, 1, 0, 1, None)
 79            self._control_transfer(0x02, 1, 0, 129, None)
 80            self._control_transfer(0x41, 2, 2, 0, None)
 81
 82            # Retrieve hardware and calibration data
 83            self._get_hardware_version()
 84            self._get_serial_number()
 85            self._get_factory_calibration()
 86            self._get_amb_measure()
 87            self._setup_measurement()
 88
 89            self.spyderData['isOpen'] = True
 90            return True
 91        except usb.core.USBError as e:
 92            print(f"USB error occurred: {str(e)}")
 93            return False

 95    def _control_transfer(self, bmRequestType, bRequest, wValue, wIndex, data_or_wLength):
 96        """
 97        Performs a USB control transfer.
 98
 99        Args:
100            bmRequestType (int): The request type.
101            bRequest (int): The specific request.
102            wValue (int): Value parameter for the request.
103            wIndex (int): Index parameter for the request.
104            data_or_wLength (int or None): Data length or None.
105
106        Returns:
107            int or None: Result of the control transfer.
108        """
109        try:
110            return self.dev.ctrl_transfer(bmRequestType, bRequest, wValue, wIndex, data_or_wLength)
111        except NotImplementedError:
112            print("Control transfer not supported on this platform.")
113            return None

115    def _bulk_transfer(self, cmd, outSize):
116        """
117        Performs a USB bulk transfer for sending commands and reading data.
118
119        Args:
120            cmd (list): List of bytes to send as a command.
121            outSize (int): Number of bytes to read as a response.
122
123        Returns:
124            numpy.ndarray: The response data read from the device.
125        """
126        try:

129        except NotImplementedError:
130            print("Bulk transfer not supported on this platform.")

133    def _get_hardware_version(self):
134        """
135        Retrieves the hardware version of the SpyderX device.
136        """
137        out = self._bulk_transfer([0xd9, 0x42, 0x33, 0x00, 0x00], 28)
138        if out is not None:
139            self.spyderData['HWvn'] = out[5:9].tobytes().decode()
140
141    def _get_serial_number(self):
142        """
143        Retrieves the serial number of the SpyderX device.
144        """
145        out = self._bulk_transfer([0xc2, 0x5c, 0x37, 0x00, 0x00], 42)
146        if out is not None:
147            self.spyderData['serNo'] = out[9:17].tobytes().decode()
148
149    def _get_factory_calibration(self):
150        """
151        Retrieves the factory calibration data, including the calibration matrix.
152        """
153        out = self._bulk_transfer([0xcb, 0x05, 0x73, 0x00, 0x01, 0x00], 47)
154        if out is not None:
155            out = out[5:]
156            matrix = np.zeros((3, 3))
157            for i in range(3):
158                for j in range(3):
159                    k = i * 3 + j
160                    matrix[i, j] = self._read_IEEE754(out[k * 4 + 4:k * 4 + 8])
161            self.spyderData['calibration'] = {'matrix': matrix}
162
163    @staticmethod
164    def _read_nORD_be(input_bytes):
165        """
166        Reads a big-endian integer from bytes.
167
168        Args:
169            input_bytes (bytes): Input bytes.
170
171        Returns:
172            int: Big-endian integer value.
173        """
174        return int.from_bytes(input_bytes, byteorder='big')
175
176    @staticmethod
177    def _read_IEEE754(input_bytes):
178        """
179        Decodes an IEEE-754 floating-point value from bytes.
180
181        Args:
182            input_bytes (bytes): Input bytes in little-endian format.
183
184        Returns:
185            float: The decoded floating-point value.
186        """
187        input_bytes = input_bytes[::-1]
188        binary = ''.join(f'{byte:08b}' for byte in input_bytes)
189        sign = int(binary[0])
190        exponent = int(binary[1:9], 2)
191        fraction = int(binary[9:], 2) / 2 ** 23
192        return (-1) ** sign * (1 + fraction) * 2 ** (exponent - 127)
193
194    def calibrate(self):
195        """
196        Performs black-level calibration of the SpyderX device.
197        """
198        if not self.spyderData.get('isOpen', False):
199            self.initialize()
200        self._control_transfer(0x41, 2, 2, 0, None)
201
202    def measure(self):
203        """
204        Measures the luminance and color values from the monitor.
205
206        Returns:
207            numpy.ndarray: The measured XYZ color values, where Y is luminance.
208        """
209        self._control_transfer(0x41, 2, 2, 0, None)
210        out = self._bulk_transfer([0xd2, 0x3f, 0xb9, 0x00, 0x07], 13)
211        if out is not None:
212            raw = np.array(struct.unpack('>HHHH', out[5:]))
213            XYZ = np.dot(raw[:3], self.spyderData['calibration']['matrix'])
214            return XYZ
215
216    def close(self):
217        """
218        Releases the USB resources and closes the connection to the SpyderX device.
219        """
220        if self.dev:
221            usb.util.dispose_resources(self.dev)
222        self.spyderData['isOpen'] = False

224def xyz_to_lms(xyz):
225    # XYZ to LMS conversion matrix (Hunt-Pointer-Estevez)
226    xyz_to_lms_matrix = np.array([
227        [0.4002, 0.7076, -0.0808],
228        [-0.2263, 1.1653, 0.0457],
229        [0.0, 0.0, 0.9182]
230    ])
231    return np.dot(xyz_to_lms_matrix, xyz)

233class GammaFitter:
234    """
235    A class to fit a gamma function to luminance data for monitor calibration.
236
237    This class takes grayscale intensity values and corresponding luminance measurements,
238    normalizes the data, and fits a gamma function to estimate the display gamma.
239    It also provides tools for visualizing the fit and customizing fit parameters.
240
241    Attributes:
242        original_intensities (numpy.ndarray): Original grayscale intensity values.
243        original_luminance (numpy.ndarray): Original luminance measurements.
244        intensities (numpy.ndarray): Normalized grayscale intensities [0, 1].
245        luminance (numpy.ndarray): Normalized luminance values [0, 1].
246        params (list): Parameters of the fitted gamma function [a, b, c].
247        gamma (float): The gamma value (b) obtained from the fitted curve.
248        lower_bounds (list): Lower bounds for curve fitting parameters.
249        upper_bounds (list): Upper bounds for curve fitting parameters.
250        initial_guess (list): Initial guess for the curve fitting parameters.
251    """
252
253    def __init__(self, intensities, luminance):
254        """
255        Initializes the GammaFitter class.
256
257        Args:
258            intensities (list or array): The grayscale intensity values.
259            luminance (list or array): The corresponding luminance measurements.
260        """
261        # Store original values for plotting
262        self.original_intensities = np.array(intensities)
263        self.original_luminance = np.array(luminance)
264
265        # Normalize input data to the range [0, 1]
266        self.intensities = (self.original_intensities - np.min(self.original_intensities)) / (
267                    np.max(self.original_intensities) - np.min(self.original_intensities))
268        self.luminance = (self.original_luminance - np.min(self.original_luminance)) / (
269                    np.max(self.original_luminance) - np.min(self.original_luminance))
270
271        self.params = None
272        self.gamma = None
273
274        # Default values for bounds and initial guess based on monitor gamma calibration
275        self.lower_bounds = [0, 0, 0]  # Lower bounds: a  0, b  1 (physical gamma), c  0
276        self.upper_bounds = [10, 5, 1]  # Upper bounds: a  10, b  5, c  0.2
277        self.initial_guess = [0, 1, 0.01]  # Typical monitor gamma: a=1, b=2.2, c=0.01
278
279    def gamma_function(self, x, a, b, c):
280        """
281        Gamma function to model luminance as a function of intensity.
282
283        Args:
284            x (float or array): The normalized intensity values.
285            a (float): Scaling factor for luminance.
286            b (float): Gamma value (exponent).
287            c (float): Offset or baseline luminance.
288
289        Returns:
290            float or array: Modeled luminance values.
291        """
292        return a * (x ** b) + c
293
294    def set_bounds(self, lower_bounds, upper_bounds):
295        """
296        Set custom lower and upper bounds for the gamma fit parameters.
297
298        Args:
299            lower_bounds (list): Lower bounds for the parameters [a, b, c].
300            upper_bounds (list): Upper bounds for the parameters [a, b, c].
301        """
302        self.lower_bounds = lower_bounds
303        self.upper_bounds = upper_bounds
304
305    def set_initial_guess(self, initial_guess):
306        """
307        Set a custom initial guess for the gamma fit parameters.
308
309        Args:
310            initial_guess (list): Initial guess for the parameters [a, b, c].
311        """
312        self.initial_guess = initial_guess
313
314    def fit(self):
315        """
316        Fits the gamma function to the normalized intensity and luminance data.
317
318        Returns:
319            list: The fitted parameters [a, b, c], where 'b' is the gamma value.
320        """
321        bounds = (self.lower_bounds, self.upper_bounds)
322        self.params, _ = curve_fit(self.gamma_function, self.intensities, self.luminance, p0=self.initial_guess,
323                                   bounds=bounds)
324        self.gamma = self.params[1]
325        return self.params
326
327    def plot(self):
328        """
329        Plots the original data and the fitted gamma curve.
330
331        Raises:
332            ValueError: If the fit has not been performed before calling this method.
333        """
334        if self.params is None:
335            raise ValueError("Fit the data first using the 'fit' method before plotting.")
336
337        plt.figure(figsize=(8, 6))
338        plt.scatter(self.original_intensities, self.original_luminance, label='Original Data', color='blue', alpha=0.6)
339
340        # Generate the fit curve in the normalized range
341        x_fit_normalized = np.linspace(0, 1, 100)
342        y_fit_normalized = self.gamma_function(x_fit_normalized, *self.params)
343
344        # Convert back to the original scale for plotting
345        x_fit_original = x_fit_normalized * (
346                    np.max(self.original_intensities) - np.min(self.original_intensities)) + np.min(
347            self.original_intensities)
348        y_fit_original = y_fit_normalized * (
349                    np.max(self.original_luminance) - np.min(self.original_luminance)) + np.min(self.original_luminance)
350
351        plt.plot(x_fit_original, y_fit_original,
352                 label=f'Gamma={self.params[1]:.3f}, Base luminance={self.params[2]:.3f}',
353                 color='red')
354        plt.xlabel('Original Intensity')
355        plt.ylabel('Original Luminance')
356        plt.title('Gamma Function Fit for Monitor Calibration')
357        plt.legend()
358        plt.grid()
359        plt.show()
